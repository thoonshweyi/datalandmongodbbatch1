$map

db.ladies.aggregate([
    {
        $project:{
            name:1,
            filterScores:{
                $map:{
                    input:"$scores",
                    as:"score",
                    in:{
                        $multiply:["$$score",2]
                    }
                }
            }
        }
    }
])

$filter
db.ladies.aggregate([
    {
        $project:{
            name:1,
            filterScores:{
                $filter:{
                    input:"$scores",
                    as:"score",
                    cond:{
                        $gte:["$$score",60]
                    }
                }
            }
        }
    }
])

$reduce

db.ladies.aggregate([
    {
        $project:{
            name:1,
            filterScores:{
                $reduce:{
                    input:"$scores",
                    initialValue: 0,
                    in:{
                        $add:["$$value","$$this"]
                    }
                }
            }
        }
    }
])
